69 void GeomBase::calcVertexNorms() {
70 
71  for (int i = 0; i < verts.size(); i++) {
72  Vector3 v;
73  for (int j = 0; j < faces.size(); j++) {
74  if (&verts.at(i) == faces.at(j)[0] || &verts.at(i) == faces.at(j)[1] ||
75  &verts.at(i) == faces.at(j)[2]) {
76  v += faces.at(j).getNorm();
77  }
78  }
79  v.normalize();
80  verts.at(i).setNormal(v);
81  }

84 void GeomBase::sortFaces() {
85  bool swapped = true;
86  int j = 0;
87  //Face3 tmp;

89  swapped = false;
90  j++;
91  for (int i = 0; i < faces.size() - j; i++) {
92  if (faces.at(i).getCentroid().z > faces.at(i + 1).getCentroid().z) {
93  Face3 tmp = faces.at(i);
94  faces.at(i) = faces.at(i + 1);
95  faces.at(i + 1) = tmp;
